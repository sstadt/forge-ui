--- title: 'fSelect' id: 'vue-fselect' index: 7 ---

Renders a select input given a set of options

Params

v-model
String; required
The selected value
label
String; defaults to an empty string
The label to render for the select input
options
Array; required
An array of options to render in the select menu; each array element should be an object with a label and value parameter
required
Boolean; defaults to false
When set to false, will render a select option with no value

selected: {{ selectedOption }}

 

<f-select v-model="selectedOption" :options="selectOptions"></f-select>


new Vue({
  el: '#select_demo',
  data: {
    selectedOption: '',
    selectOptions: [
      { label: 'One Option', value: '1' },
      { label: 'Two Option', value: '2' },
      { label: 'Red Option', value: 'Red' },
      { label: 'Blue Option', value: 'Blue' }
    ]
  }
});
