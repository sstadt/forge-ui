--- title: 'Modal' id: 'vue-modal' index: 3 ---

A simple modal dialog.

Params

modalSize
String
Sets the size of the modal. Can be set to 'small' or 'full' to increase or decrease the size of the modal.

Methods

open()
Open the dialog.
close()
Close the dialog.

Look, a Modal!

some super awesome modal content

 

<button type="button" class="button" @click="openModal()">Open Modal</button>

<modal ref="myModal">
  <div slot="header">
    <h3>Look, a Modal!</h3>
  </div>
  <div slot="content">
    <p>some super awesome modal content</p>
    <button type="button" class="button" @click="closeModal()">Close</button>
  </div>
</modal>


new Vue({
  el: '#modal_demo',
  methods: {
    openModal() {
      this.$refs.myModal.open();
    },
    closeModal() {
      this.$refs.myModal.close();
    }
  }
});
