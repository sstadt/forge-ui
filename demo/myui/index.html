Look, a Modal!

some super awesome modal content

 

<button type="button" class="button" v-on:click="showModal = true">Open Modal</button>

<modal :show.sync="showModal">
  <div slot="header">
    <h3>Look, a Modal!</h3>
  </div>
  <div slot="content">
    <p>some super awesome modal content</p>
    <button type="button" class="button" v-on:click="showModal = false">Close</button>
  </div>
</modal>


new Vue({
  el: '#modal_demo',
  data: {
    showModal: false
  }
});
